pragma solidity ^0.8.0;

contract EmployeeDetails {
    struct Employee {
        uint id;
        string name;
        uint salary;
        string joiningDate;
    }

    Employee[] public employees;

    // Add new employee
    function addEmployee(uint _id, string memory _name, uint _salary, string memory _date) public {
        employees.push(Employee(_id, _name, _salary, _date));
    }

    // Display employee details
    function getEmployee(uint index) public view returns (uint, string memory, uint, string memory) {
        require(index < employees.length, "Invalid index");
        Employee memory e = employees[index];
        return (e.id, e.name, e.salary, e.joiningDate);
    }
}
